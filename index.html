// 计算斐波那契数列 —— 递归如此简单
func fib(n) {
    if (n <= 1) { return n; }
    return fib(n-1) + fib(n-2);
}

var i = 0;
while (i < 10) {
    print "fib(", i, ") = ", fib(i), "\n";
    i = i + 1;
}
